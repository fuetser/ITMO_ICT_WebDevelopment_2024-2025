    Страница регистрации

    Описание

    На данной странице пользователь может зарегистрироваться, указав следующие данные:

    1. Имя пользователя
    2. Адрес электронной почты
    3. Пароль

    Код формы

    <template>
  <v-container class="d-flex justify-center align-center">
    <v-card elevation="10" class="pa-5" width="30%" min-width="400px">
      <v-card-title>Регистрация</v-card-title>
      <v-card-text>
        <v-alert
          v-if="errorMessage"
          type="error"
          dismissible
          class="mb-4"
          @click:close="clearError"
        >
          {{ errorMessage }}
        </v-alert>
        <v-form ref="formRef" v-model="valid" @submit.prevent="submitForm">
          <v-text-field
            v-model="form.name"
            label="Имя пользователя"
            :rules="[rules.required, rules.minLength(4)]"
            required
          ></v-text-field>
          <v-text-field
            v-model="form.email"
            label="Email"
            :rules="[rules.required, rules.email]"
            required
          ></v-text-field>
          <v-text-field
            v-model="form.password"
            label="Пароль"
            :type="showPassword ? 'text' : 'password'"
            :append-inner-icon="showPassword ? 'mdi-eye-off' : 'mdi-eye'"
            @click:append-inner="togglePasswordVisibility"
            :rules="[rules.required, rules.minLength(6)]"
            required
          ></v-text-field>
          <v-text-field
            v-model="form.confirmPassword"
            label="Подтвердите пароль"
            :type="showConfirmPassword ? 'text' : 'password'"
            :append-inner-icon="showConfirmPassword ? 'mdi-eye-off' : 'mdi-eye'"
            @click:append-inner="toggleConfirmPasswordVisibility"
            :rules="[rules.required, rules.matchPassword]"
            required
          ></v-text-field>
          <v-btn :disabled="!valid" type="submit" color="primary" block>
            Зарегистрироваться
          </v-btn>
          <v-btn class="mt-6" color="secondary" block @click="redirectToLogin">
            Войти
          </v-btn>
        </v-form>
      </v-card-text>
    </v-card>
  </v-container>
</template>

<script setup>
import axios from "axios";
import { reactive, ref } from "vue";
import { useRouter } from "vue-router";

const router = useRouter();
const redirectToLogin = () => {
  router.push("/login");
};


const valid = ref(false);
const formRef = ref(null);
const showPassword = ref(false);
const showConfirmPassword = ref(false);
const errorMessage = ref("");
const form = reactive({
  name: "",
  email: "",
  password: "",
  confirmPassword: "",
});

const rules = {
  required: (value) => !!value || "Обязательное поле",
  email: (value) =>
    /^[\w-.]+@[\w-]+\.+[\w-]{2,4}$/.test(value) || "Некорректный адрес",
  minLength: (length) => (value) =>
    value.length >= length || `Минимальная длина ${length} символов`,
  matchPassword: (value) =>
    value === form.password || "Пароли должны совпадать",
};

const togglePasswordVisibility = () =>
  (showPassword.value = !showPassword.value);
const toggleConfirmPasswordVisibility = () =>
  (showConfirmPassword.value = !showConfirmPassword.value);

const clearError = () => {
  errorMessage.value = "";
};

const submitForm = async () => {
  const { valid } = await formRef.value.validate();
  if (valid) {
    const registrationData = {
      password: form.password,
      username: form.name,
      email: form.email,
    };
    axios.defaults.headers.common["Authorization"] = null;
    try {
      const response = await axios.post(
        "http://localhost:8000/auth/users/",
        registrationData
      );
      if (response.status === 201) {
        redirectToLogin();
      }
    } catch (err) {
      errorMessage.value = "Произошла ошибка регистрации";
    }
  }
};
</script>

    Код верхней панели

    <template>
  <v-app-bar :elevation="2">
    <template v-slot:prepend>
      <v-app-bar-nav-icon
        @click="showDrawer = !showDrawer"
      ></v-app-bar-nav-icon>
    </template>

    <v-app-bar-title>Head Teacher App</v-app-bar-title>
    <div v-if="isUserAuthenticated()">
      <v-btn color="primary" @click="navigateToPage('/login')">Вход</v-btn>
      <v-btn color="primary" @click="navigateToPage('/registration')"
        >Регистрация</v-btn
      >
    </div>
    <div v-else>
      <v-btn color="primary" @click="logout">Выйти</v-btn>
    </div>
  </v-app-bar>
  <v-navigation-drawer
    v-model="showDrawer"
    :location="mobile ? 'bottom' : undefined"
    temporary
  >
    <v-list>
      <v-list-item
        v-for="navigationItem in navigationItems"
        :key="navigationItem.value"
        :title="navigationItem.title"
        @click="navigateToPage(navigationItem.value)"
      ></v-list-item>
    </v-list>
  </v-navigation-drawer>
</template>

<script setup>
import axios from "axios";
import { ref } from "vue";
import { useDisplay } from "vuetify";
import { useRouter } from "vue-router";

const { mobile } = useDisplay();
const navigationItems = [
  { title: "Классы", value: "/classes" },
  { title: "Кабинеты", value: "/rooms" },
  { title: "Расписание", value: "/" },
  { title: "Ученики", value: "/students" },
  { title: "Предметы", value: "/subjects" },
  { title: "Учителя", value: "/teachers" },
];

const isUserAuthenticated = () => !localStorage.getItem("token");

const showDrawer = ref(false);

const router = useRouter();

const navigateToPage = (path) => {
  router.push(path);
};

const logout = async () => {
  const token = localStorage.getItem("token");
  axios.defaults.headers.common["Authorization"] = `Token ${token}`;
  try {
    const response = await axios.post("http://localhost:8000/auth/token/logout/");
    if (response.status === 204) {
      localStorage.removeItem("token");
      navigateToPage("/login");
    }
  } catch (err) {
    console.error("Error fetching classes:", err);
  }
};
</script>
